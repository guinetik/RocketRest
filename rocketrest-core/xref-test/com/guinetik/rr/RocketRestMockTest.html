30      @Before
31      public void setUp() {
32          // Create a basic configuration
33          RocketRestConfig config = RocketRestConfig.builder(BASE_URL).build();
34          mockClient = new RocketRestMock(config);
35  
36          // Set up mock responses
37          mockClient.addMockResponse(HttpConstants.Methods.GET, SUCCESS_ENDPOINT, (url, body) -> "Success Response");
38          mockClient.addMockResponse(HttpConstants.Methods.POST, SUCCESS_ENDPOINT, (url, body) -> "Post Success");
39          mockClient.addMockResponse(HttpConstants.Methods.GET, ERROR_ENDPOINT, (url, body) -> {
40              throw new ApiException("Internal Server Error", "Server Error Response", "Internal Server Error", 
41                  HttpConstants.StatusCodes.INTERNAL_SERVER_ERROR);
42          });
43          mockClient.addMockResponse(HttpConstants.Methods.GET, REGEX_ENDPOINT, (url, body) -> "User Response", true);
44      }

89      @Test
90      public void testInvocationCounting() {
91          // Make some requests
92          mockClient.get(SUCCESS_ENDPOINT, String.class);
93          mockClient.get(SUCCESS_ENDPOINT, String.class);
94          mockClient.post(SUCCESS_ENDPOINT, String.class);
95  
96          // Verify invocation counts
97          assertEquals(2, mockClient.getInvocationCount(HttpConstants.Methods.GET, SUCCESS_ENDPOINT));
98          assertEquals(1, mockClient.getInvocationCount(HttpConstants.Methods.POST, SUCCESS_ENDPOINT));
99          assertEquals(0, mockClient.getInvocationCount(HttpConstants.Methods.GET, "/nonexistent"));
100 
101         // Test regex pattern invocation counting
102         mockClient.get("/users/123", String.class);
103         mockClient.get("/users/456", String.class);
104         assertEquals(2, mockClient.getInvocationCount(HttpConstants.Methods.GET, REGEX_ENDPOINT));
105 
106         // Test reset
107         mockClient.resetInvocationCounts();
108         assertEquals(0, mockClient.getInvocationCount(HttpConstants.Methods.GET, SUCCESS_ENDPOINT));
109         assertEquals(0, mockClient.getInvocationCount(HttpConstants.Methods.POST, SUCCESS_ENDPOINT));
110         assertEquals(0, mockClient.getInvocationCount(HttpConstants.Methods.GET, REGEX_ENDPOINT));
111     }

113     @Test
114     public void testLatencySimulation() {
115         // Add latency for a specific endpoint
116         long latencyMs = HttpConstants.Timeouts.QUICK_TIMEOUT;
117         mockClient.withLatency(LATENCY_ENDPOINT, latencyMs);
118         mockClient.addMockResponse(HttpConstants.Methods.GET, LATENCY_ENDPOINT, (url, body) -> "Delayed Response");
119 
120         // Measure execution time
121         long startTime = System.nanoTime();
122         String response = mockClient.get(LATENCY_ENDPOINT, String.class);
123         long endTime = System.nanoTime();
124         long durationMs = TimeUnit.NANOSECONDS.toMillis(endTime - startTime);
125 
126         // Verify response and timing
127         assertEquals("Delayed Response", response);
128         assertTrue("Request should take at least the configured latency",
129                 durationMs >= latencyMs);
130     }

168     @Test
169     public void testMultipleLatencyPatterns() {
170         // Add different latencies for different patterns
171         mockClient.withLatency("/slow/.*", HttpConstants.Timeouts.QUICK_TIMEOUT);
172         mockClient.withLatency("/very/slow/.*", HttpConstants.Timeouts.DEFAULT_CONNECT_TIMEOUT);
173         mockClient.addMockResponse(HttpConstants.Methods.GET, "/slow/test", (url, body) -> "Slow Response");
174         mockClient.addMockResponse(HttpConstants.Methods.GET, "/very/slow/test", (url, body) -> "Very Slow Response");
175 
176         // Test first pattern
177         long startTime = System.nanoTime();
178         String response = mockClient.get("/slow/test", String.class);
179         long durationMs = TimeUnit.NANOSECONDS.toMillis(System.nanoTime() - startTime);
180         assertEquals("Slow Response", response);
181         assertTrue("Should take at least 100ms", durationMs >= 100);
182 
183         // Test second pattern
184         startTime = System.nanoTime();
185         response = mockClient.get("/very/slow/test", String.class);
186         durationMs = TimeUnit.NANOSECONDS.toMillis(System.nanoTime() - startTime);
187         assertEquals("Very Slow Response", response);
188         assertTrue("Should take at least 200ms", durationMs >= HttpConstants.Timeouts.DEFAULT_CONNECT_TIMEOUT);
189     }
