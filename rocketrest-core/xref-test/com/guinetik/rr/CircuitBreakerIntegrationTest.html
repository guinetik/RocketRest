22  @Ignore("Integration tests - depends on external service. Run explicitly with -Dtest=CircuitBreakerIntegrationTest")

25      private RocketRest client;
26      private static final String API_BASE_URL = "https://tools-httpstatus.pickup-services.com/";
27      private static final int FAILURE_THRESHOLD = 3;
28      private static final long RESET_TIMEOUT_MS = 2000; // 2 seconds for faster testing

30      @Before
31      public void setUp() {
32          // Create configuration for the API with circuit breaker settings
33          RocketRestConfig config = RocketRestConfig.builder(API_BASE_URL)
34                  .defaultOptions(options -> {
35                      // Configure circuit breaker options
36                      options.set(HttpConstants.CircuitBreaker.CIRCUIT_BREAKER_ENABLED, true);
37                      options.set(HttpConstants.CircuitBreaker.CIRCUIT_BREAKER_FAILURE_THRESHOLD, FAILURE_THRESHOLD);
38                      options.set(HttpConstants.CircuitBreaker.CIRCUIT_BREAKER_RESET_TIMEOUT_MS, RESET_TIMEOUT_MS);
39                      options.set(HttpConstants.CircuitBreaker.CIRCUIT_BREAKER_FAILURE_POLICY, HttpConstants.CircuitBreaker.CIRCUIT_BREAKER_POLICY_SERVER_ONLY);
40                  })
41                  .build();
42          
43          // Create a real client using the config
44          client = new RocketRest(config);
45      }

61      @Test
62      public void testCircuitBreakerOpens() {
63          // First, make a successful call
64          String successResponse = client.get("/200", String.class);
65          assertNotNull("Success response should not be null", successResponse);
66          
67          // Now trigger enough failures to open the circuit
68          for (int i = 0; i < FAILURE_THRESHOLD - 1; i++) {
69              try {
70                  // Request 500 status to trigger errors
71                  client.get("/500", String.class);
72                  // We should still get a response but as an error
73              } catch (Exception e) {
74                  // Expected exception, but should not be circuit breaker yet
75                  assertFalse("Should not be circuit breaker exception on failure #" + (i+1),
76                          e instanceof CircuitBreakerOpenException);
77              }
78          }
79          
80          // The last failure should open the circuit
81          try {
82              client.get("/500", String.class);
83              fail("Should have thrown CircuitBreakerOpenException");
84          } catch (Exception e) {
85              // Verify the correct exception type
86              assertTrue("Expected CircuitBreakerOpenException but got: " + e.getClass().getName(), 
87                      e instanceof CircuitBreakerOpenException);
88              
89              // Verify the exception contains timing information
90              CircuitBreakerOpenException cbException = (CircuitBreakerOpenException) e;
91              assertTrue("Should have time until reset", cbException.getEstimatedMillisUntilReset() > 0);
92          }
93          
94          // Now the circuit should be open, any further calls should throw CircuitBreakerOpenException
95          try {
96              client.get("/200", String.class);
97              fail("Should have thrown CircuitBreakerOpenException");
98          } catch (Exception e) {
99              // Verify the correct exception type
100             assertTrue("Expected CircuitBreakerOpenException but got: " + e.getClass().getName(), 
101                     e instanceof CircuitBreakerOpenException);
102             
103             // Verify the exception contains timing information
104             CircuitBreakerOpenException cbException = (CircuitBreakerOpenException) e;
105             assertTrue("Should have time until reset", cbException.getEstimatedMillisUntilReset() > 0);
106         }
107     }

109     @Test
110     public void testCircuitBreakerWithFluentAPI() {
111         // Trigger failures to reach (but not exceed) the threshold
112         for (int i = 0; i < FAILURE_THRESHOLD - 1; i++) {
113             Result<String, ApiError> result = client.fluent().get("/500", String.class);
114             assertTrue("Should be a failure result for 500 status", result.isFailure());
115             
116             ApiError error = result.getError();
117             assertEquals("Should be HTTP_ERROR error type on request #" + (i+1),
118                     ApiError.ErrorType.HTTP_ERROR, error.getErrorType());
119             assertEquals("Status code should be 500 on request #" + (i+1),
120                     HttpConstants.StatusCodes.INTERNAL_SERVER_ERROR, error.getStatusCode());
121         }
122         
123         // The last failure might either be HTTP_ERROR or CIRCUIT_OPEN
124         // depending on timing and exactly when the circuit opens
125         Result<String, ApiError> lastFailureResult = client.fluent().get("/500", String.class);
126         assertTrue("Should be a failure result", lastFailureResult.isFailure());
127         
128         // After reaching FAILURE_THRESHOLD, make sure circuit is definitely open
129         // with a small delay to ensure circuit breaker state propagation
130         try {
131             Thread.sleep(100);
132         } catch (InterruptedException e) {
133             Thread.currentThread().interrupt();
134         }
135         
136         // Now the circuit should definitely be open
137         Result<String, ApiError> circuitOpenResult = client.fluent().get("/200", String.class);
138         
139         assertTrue("Should be a failure result due to open circuit", circuitOpenResult.isFailure());
140         ApiError error = circuitOpenResult.getError();
141         assertNotNull("Error should not be null", error);
142         assertEquals("Should be CIRCUIT_OPEN error type", 
143                 ApiError.ErrorType.CIRCUIT_OPEN, error.getErrorType());
144         // CircuitBreakerOpenException errors don't have a status code (it's 0)
145         assertEquals("Circuit breaker errors should have status code 0", 0, error.getStatusCode());
146     }

152     @Test
153     public void testCircuitBreakerResetsAfterTimeout() throws Exception {
154         // Trigger enough failures to open the circuit
155         for (int i = 0; i < FAILURE_THRESHOLD; i++) {
156             try {
157                 client.get("/500", String.class);
158             } catch (Exception e) {
159                 // Expected exceptions from 500 responses
160             }
161         }
162         
163         // Verify the circuit is open
164         try {
165             client.get("/200", String.class);
166             fail("Circuit should be open");
167         } catch (CircuitBreakerOpenException e) {
168             // Expected - circuit is open
169             System.out.println("Circuit breaker is open as expected. Waiting for reset...");
170         }
171         
172         // Wait for the reset timeout
173         System.out.println("Waiting " + RESET_TIMEOUT_MS + "ms for circuit reset...");
174         Thread.sleep(RESET_TIMEOUT_MS + 1000); // Add a bit extra for safety
175         
176         // Now the circuit should be half-open, allowing one test request
177         try {
178             String response = client.get("/200", String.class);
179             System.out.println("Circuit is half-open or closed. Successful response: " + response);
180             assertNotNull("Response should not be null after reset", response);
181             
182             // Another successful call should work as the circuit is now closed
183             response = client.get("/200", String.class);
184             System.out.println("Circuit is fully closed. Successful response: " + response);
185             assertNotNull("Second response should not be null", response);
186         } catch (CircuitBreakerOpenException e) {
187             fail("Circuit should have reset after timeout, but got: " + e.getMessage());
188         }
189     }
