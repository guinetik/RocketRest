13      private final NoAuthStrategy strategy = new NoAuthStrategy();
14  
15      @Test
16      public void testApplyAuthHeadersDoesNotModifyHeaders() {
17          RocketHeaders headers = RocketHeaders.defaultJson();
18          RocketHeaders result = strategy.applyAuthHeaders(headers);
19          assertSame("Should return same headers instance", headers, result);
20          assertFalse("Authorization header should not be present", result.contains(RocketHeaders.Names.AUTHORIZATION));
21      }
22  
23      @Test
24      public void testNeedsRefresh() {
25          assertFalse(strategy.needsTokenRefresh());
26      }
27  
28      @Test
29      public void testRefreshCredentialsAlwaysTrue() {
30          assertTrue(strategy.refreshCredentials());
31      }
