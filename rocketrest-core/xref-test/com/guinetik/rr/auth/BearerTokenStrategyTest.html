10  /**
11   * Unit tests for {@link BearerTokenStrategy}.

13  public class BearerTokenStrategyTest {
14  
15      private static final String TOKEN = "abc123";
16  
17      @Test
18      public void testApplyAuthHeadersSetsBearerAuthorization() {
19          BearerTokenStrategy strategy = new BearerTokenStrategy(TOKEN);
20          RocketHeaders headers = new RocketHeaders();
21          strategy.applyAuthHeaders(headers);
22          assertEquals("Bearer " + TOKEN, headers.get(RocketHeaders.Names.AUTHORIZATION));
23      }
24  
25      @Test
26      public void testRefreshCredentialsDefaultReturnsFalse() {
27          BearerTokenStrategy strategy = new BearerTokenStrategy(TOKEN);
28          assertFalse(strategy.refreshCredentials());
29      }
30  
31      @Test
32      public void testRefreshCredentialsCustomLogic() {
33          AtomicBoolean called = new AtomicBoolean(false);
34          BearerTokenStrategy strategy = new BearerTokenStrategy(TOKEN, () -> {
35              called.set(true);
36              return true;
37          });
38          assertTrue(strategy.refreshCredentials());
39          assertTrue("Custom supplier should have been called", called.get());
40      }
41  }
