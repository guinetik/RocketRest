122     public void testPrepareTokenRequestParamsWithRefreshToken() throws Exception {
123         // Set a refresh token via reflection
124         Field refreshTokenField = OAuth2PasswordStrategy.class.getDeclaredField("refreshToken");
125         refreshTokenField.setAccessible(true);
126         refreshTokenField.set(strategy, REFRESH_TOKEN);
127         
128         Method prepareMethod = OAuth2PasswordStrategy.class.getDeclaredMethod("prepareTokenRequestParams");
129         prepareMethod.setAccessible(true);
130         
131         Map<String, String> params = (Map<String, String>) prepareMethod.invoke(strategy);
132         
133         assertEquals("refresh_token", params.get("grant_type"));
134         assertEquals(REFRESH_TOKEN, params.get("refresh_token"));
135         assertEquals(CLIENT_ID, params.get("client_id"));
136         assertEquals(CLIENT_SECRET, params.get("client_secret"));
137         // Should not include username/password when using refresh token
138         assertFalse(params.containsKey("username"));
139         assertFalse(params.containsKey("password"));
140     }

180     @Test
181     public void testNeedsTokenRefresh() throws Exception {
182         // Initially should need refresh (no token)
183         assertTrue(strategy.needsTokenRefresh());
184         
185         // Set a token but no expiry - should still need refresh
186         Field accessTokenField = AbstractOAuth2Strategy.class.getDeclaredField("accessToken");
187         accessTokenField.setAccessible(true);
188         accessTokenField.set(strategy, "test-token");
189         assertTrue(strategy.needsTokenRefresh());
190         
191         // Set expiry in the future - should not need refresh
192         Field expiryField = AbstractOAuth2Strategy.class.getDeclaredField("tokenExpiryTime");
193         expiryField.setAccessible(true);
194         // Set expiry to 1 hour from now
195         expiryField.set(strategy, Date.from(Instant.now().plusSeconds(3600)));
196         assertFalse(strategy.needsTokenRefresh());
197         
198         // Set expiry to near future (< 5 min) - should need refresh
199         expiryField.set(strategy, Date.from(Instant.now().plusSeconds(200)));
200         assertTrue(strategy.needsTokenRefresh());
201     }
