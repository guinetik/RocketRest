83      @Test
84      public void testGetKeys() {
85          int initialCount = 0;
86          for (String key : options.getKeys()) {
87              initialCount++;
88          }
89  
90          options.set("key1", "value1");
91          options.set("key2", "value2");
92          options.set("key3", "value3");
93  
94          Iterable<String> keys = options.getKeys();
95          int count = 0;
96          boolean hasKey1 = false, hasKey2 = false, hasKey3 = false;
97          for (String key : keys) {
98              count++;
99              if (key.equals("key1")) hasKey1 = true;
100             if (key.equals("key2")) hasKey2 = true;
101             if (key.equals("key3")) hasKey3 = true;
102         }
103         assertTrue(hasKey1 && hasKey2 && hasKey3);
104         assertEquals(initialCount + 3, count);
105     }
