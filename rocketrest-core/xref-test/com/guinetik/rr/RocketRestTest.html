65      @Test
66      public void testSyncApiMethods() throws Exception {
67          // Set up the mock API endpoint
68          client.setBaseUrl(MOCK_API_URL);
69          
70          // 1. Get all records and verify count
71          String allRecords = client.sync().get("/records", String.class);
72          assertNotNull("All records response should not be null", allRecords);
73          List<Map<String, Object>> records = objectMapper.readValue(allRecords, List.class);
74          assertEquals("Should have 5 records initially", 5, records.size());
75          
76          // 2. Verify first record name is Guinetik
77          Map<String, Object> firstRecord = records.get(0);
78          assertEquals("First record should have name Guinetik", "Guinetik", firstRecord.get("name"));
79          
80          // 3. Add new record
81          Map<String, String> newRecord = new HashMap<>();
82          newRecord.put("name", "Hello from Integrated Test");
83          
84          String postResponse = client.sync().post("/records", newRecord, String.class);
85          assertNotNull("Post response should not be null", postResponse);
86          Map<String, Object> postedData = objectMapper.readValue(postResponse, Map.class);
87          String newRecordId = postedData.get("id").toString();
88          assertEquals("Posted record should have correct name", "Hello from Integrated Test", postedData.get("name"));
89          
90          // 4. Update the new record
91          Map<String, String> updatedRecord = new HashMap<>();
92          updatedRecord.put("name", "Updated Name");
93          
94          String putResponse = client.sync().put("/records/" + newRecordId, updatedRecord, String.class);
95          assertNotNull("Put response should not be null", putResponse);
96          Map<String, Object> updatedData = objectMapper.readValue(putResponse, Map.class);
97          assertEquals("Updated record should have new name", "Updated Name", updatedData.get("name"));
98          
99          // 5. Delete the record
100         String deleteResponse = client.sync().delete("/records/" + newRecordId, String.class);
101         assertNotNull("Delete response should not be null", deleteResponse);
102         
103         // 6. Get all records again and verify the count
104         String finalRecords = client.sync().get("/records", String.class);
105         assertNotNull("Final records response should not be null", finalRecords);
106         List<Map<String, Object>> finalRecordsList = objectMapper.readValue(finalRecords, List.class);
107         assertEquals("Should have 5 records after deletion", 5, finalRecordsList.size());
108     }
