106     @Test
107     public void testMerge() {
108         RocketHeaders base = new RocketHeaders()
109                 .set("X-Base", "base-value")
110                 .set("X-Override", "original");
111 
112         RocketHeaders override = new RocketHeaders()
113                 .set("X-Override", "new-value")
114                 .set("X-New", "new");
115 
116         RocketHeaders merged = base.merge(override);
117 
118         assertEquals("base-value", merged.get("X-Base"));
119         assertEquals("new-value", merged.get("X-Override"));
120         assertEquals("new", merged.get("X-New"));
121     }

123     @Test
124     public void testMergeDoesNotModifyOriginal() {
125         RocketHeaders original = new RocketHeaders().set("X-Original", "value");
126         RocketHeaders other = new RocketHeaders().set("X-Other", "value");
127 
128         RocketHeaders merged = original.merge(other);
129 
130         // Original should not have X-Other
131         assertFalse(original.contains("X-Other"));
132         assertTrue(merged.contains("X-Other"));
133     }
